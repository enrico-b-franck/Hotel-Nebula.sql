
CREATE DATABASE hotel_nebula;
USE hotel_nebula;

CREATE TABLE hospedes (
    id_hospede INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    telefone VARCHAR(20),
    email VARCHAR(100),
    data_nascimento DATE
);

CREATE TABLE quartos (
    id_quarto INT PRIMARY KEY AUTO_INCREMENT,
    numero INT UNIQUE NOT NULL,
    categoria VARCHAR(50),
    capacidade INT,
    valor_diaria DECIMAL(10,2),
    status VARCHAR(30)
);

CREATE TABLE funcionarios (
    id_funcionario INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    cargo VARCHAR(50),
    telefone VARCHAR(20),
    salario DECIMAL(10,2)
);

CREATE TABLE reservas (
    id_reserva INT PRIMARY KEY AUTO_INCREMENT,
    id_hospede INT,
    id_quarto INT,
    data_checkin DATE,
    data_checkout DATE,
    status VARCHAR(30),

    FOREIGN KEY (id_hospede)
        REFERENCES hospedes(id_hospede),

    FOREIGN KEY (id_quarto)
        REFERENCES quartos(id_quarto)
);

CREATE TABLE hospedagens (
    id_hospedagem INT PRIMARY KEY AUTO_INCREMENT,
    id_reserva INT UNIQUE,
    id_funcionario INT,
    checkin_real DATETIME,
    checkout_real DATETIME,

    FOREIGN KEY (id_reserva)
        REFERENCES reservas(id_reserva),

    FOREIGN KEY (id_funcionario)
        REFERENCES funcionarios(id_funcionario)
);

CREATE TABLE pagamentos (
    id_pagamento INT PRIMARY KEY AUTO_INCREMENT,
    id_reserva INT,
    valor DECIMAL(10,2),
    data_pagamento DATE,
    metodo VARCHAR(50),
    status VARCHAR(30),

    FOREIGN KEY (id_reserva)
        REFERENCES reservas(id_reserva)
);

CREATE TABLE servicos (
    id_servico INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100),
    valor DECIMAL(10,2)
);

CREATE TABLE hospedagem_servicos (
    id_hospedagem INT,
    id_servico INT,

    PRIMARY KEY (id_hospedagem, id_servico),

    FOREIGN KEY (id_hospedagem)
        REFERENCES hospedagens(id_hospedagem),

    FOREIGN KEY (id_servico)
        REFERENCES servicos(id_servico)
);

CREATE TABLE avaliacoes (
    id_avaliacao INT PRIMARY KEY AUTO_INCREMENT,
    id_hospedagem INT,
    nota INT CHECK (nota BETWEEN 1 AND 5),
    comentario TEXT,
    data_avaliacao DATE,

    FOREIGN KEY (id_hospedagem)
        REFERENCES hospedagens(id_hospedagem)
);

-- 1. Quartos disponíveis em determinado período
SELECT *
FROM quartos q
WHERE q.id_quarto NOT IN (
    SELECT r.id_quarto
    FROM reservas r
    WHERE '2026-06-03' BETWEEN r.data_checkin AND r.data_checkout
);
-- 2. Hóspedes que mais realizaram reservas
SELECT h.nome,
       COUNT(r.id_reserva) AS total_reservas
FROM hospedes h
JOIN reservas r
ON h.id_hospede = r.id_hospede
GROUP BY h.nome
ORDER BY total_reservas DESC;

-- 3. Faturamento por mês

SELECT MONTH(data_pagamento) AS mes,
       YEAR(data_pagamento) AS ano,
       SUM(valor) AS faturamento
FROM pagamentos
WHERE status = 'Pago'
GROUP BY ano, mes
ORDER BY ano, mes;

-- 4. Serviços extras mais consumidos

SELECT s.nome,
       COUNT(*) AS quantidade
FROM servicos s
JOIN hospedagem_servicos hs
ON s.id_servico = hs.id_servico
GROUP BY s.nome
ORDER BY quantidade DESC;

-- 5. Quartos com melhores avaliações

SELECT q.numero,
       AVG(a.nota) AS media_avaliacao
FROM quartos q
JOIN reservas r
ON q.id_quarto = r.id_quarto
JOIN hospedagens h
ON r.id_reserva = h.id_reserva
JOIN avaliacoes a
ON h.id_hospedagem = a.id_hospedagem
GROUP BY q.numero
ORDER BY media_avaliacao DESC;

-- 6. Reservas canceladas
SELECT *
FROM reservas
WHERE status = 'Cancelada';

-- 7. Quantas hospedagens cada profissional atendeu
SELECT f.nome,
       COUNT(h.id_hospedagem) AS total_hospedagens
FROM funcionarios f
JOIN hospedagens h
ON f.id_funcionario = h.id_funcionario
GROUP BY f.nome
ORDER BY total_hospedagens DESC;
